import { Calendar, momentLocalizer } from 'react-big-calendar';

import moment from 'moment';
const localizer = momentLocalizer(moment);

const [recipeData, setRecipeData] = useState([]);



import moment from 'moment';

// Formatea las fechas de vencimiento en el formato adecuado
const formattedEvents = recipeData.map((recipe) => ({
  start: moment(recipe.fecha_vencimiento).toDate(),
  end: moment(recipe.fecha_vencimiento).toDate(),
  title: 'Receta vencida',
}));

  localizer={localizer}
  events={formattedEvents}
  startAccessor="start"
  endAccessor="end"
  titleAccessor="title"
/>
